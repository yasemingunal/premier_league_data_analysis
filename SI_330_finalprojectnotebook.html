team_wins_dict = {}

for idx, row in epl_db_df.iterrows():
    if row['FTR'] == 'H':
        if row['HomeTeam'] not in team_wins_dict.keys():
            team_wins_dict[row['HomeTeam']] = 1
        else: 
            team_wins_dict[row['HomeTeam']] += 1
    elif row['FTR'] == 'A':
        if row['AwayTeam'] not in team_wins_dict.keys():
            team_wins_dict[row['AwayTeam']] = 1
        else: 
            team_wins_dict[row['AwayTeam']] += 1

# print(team_wins_dict)
